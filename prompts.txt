def parse_show_interfaces_terse(text: str) -> Dict[str, Any]:
    cmd = 'show interface terse | no-more'
    try:
        result = ShowInterfacesTerse()
        lines = text.strip().splitlines()
        if not lines:
            result.total_interfaces = 0
            return asdict(result)

        # Known protocol keywords we might see in the "Proto" column
        proto_keywords = {
            "inet", "inet6", "iso", "mpls", "aenet", "tnp", "bridge", "multiservice"
        }

        last_entry = None

        # Skip header (first line typically: "Interface Admin Link Proto Local Remote")
        for raw in lines[1:]:
            line = raw.rstrip()
            if not line:
                continue

            parts = line.split()
            if not parts:
                continue

            # If a line doesn't have at least 3 tokens, it cannot be a new interface row.
            # Treat it as a continuation (e.g., "inet6", "mpls", "multiservice", arrow targets).
            if len(parts) < 3:
                # Continuation line: attempt to attach proto/local/remote to last_entry if sensible
                if last_entry and parts:
                    # Example continuation:
                    #    inet6
                    #    multiservice
                    # We set proto only if it wasn't already set.
                    if parts[0] in proto_keywords and not getattr(last_entry, "proto", None):
                        last_entry.proto = parts[0]
                continue

            # From here, assume new interface row: interface admin link [others...]
            iface, admin, link = parts[0], parts[1], parts[2]
            entry = InterfaceEntry(
                interface=iface,
                admin=admin,
                link=link,
            )

            # Parse optional columns: proto, local, remote with optional "-->" token
            # Remaining tokens start at index 3
            idx = 3
            if idx < len(parts):
                token = parts[idx]

                # Heuristic: If token looks like a protocol keyword, use it as proto
                # Otherwise, some platforms may omit the proto when no protocol is present.
                if token in proto_keywords:
                    entry.proto = token
                    idx += 1
                else:
                    # If it's not a proto keyword, keep proto as None and try to parse as local/remote if meaningful.
                    entry.proto = entry.proto if hasattr(entry, "proto") else None

                # After proto (if any), try parse local and remote, including arrow forms.
                # Patterns to support:
                #  - "<local> --> <remote>"
                #  - "--> <remote>" (no local)
                #  - "<local>" only
                #  - just "multiservice"/"mpls"/"inet6" on the same line (no local/remote)
                if idx < len(parts):
                    # Case: arrow without local, e.g., "aenet --> ae10.0"
                    if parts[idx] == "-->":
                        if idx + 1 < len(parts):
                            entry.remote = parts[idx + 1]
                    else:
                        # Assume local is next token
                        entry.local = parts[idx]
                        idx += 1
                        # Maybe arrow follows for remote
                        if idx < len(parts) and parts[idx] == "-->":
                            if idx + 1 < len(parts):
                                entry.remote = parts[idx + 1]

            result.interfaces.append(entry)
            last_entry = entry

        result.total_interfaces = len(result.interfaces)
